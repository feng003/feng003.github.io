2、隐藏内部实现

function doSth(a)
{
    b = a + doSthElse( a * 2);        
}
function doSthElse(a)
{
    return a-1;
}
var b;
doSth(2);

隐藏之后

function doSth(a)
{
        function doSthElse(a)
        {
            return a-1;
        }

        var b;

        b = a + doSthElse(a * 2);
        console.log(b * 3);
}
doSth(2);

b 和 doSthElse() 无法从外部访问。
